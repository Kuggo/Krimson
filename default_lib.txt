object __bool__ {
    func bool __or__(bool me, bool other) {
        return me || other
    }

    func bool __and__(bool me, bool other) {
        return me other
    }

    func bool __xor__(bool me, bool other) {
        return me ^ other
    }

    func bool __not__(bool me) {
        return !me
    }
}

object __fixed__<uint_> {
    func fixed __add__(fixed<uint_> me, fixed<uint_> other) {
        return me + other
    }

    func fixed __sub__(fixed<uint_> me, fixed<uint_> other) {
        return me - other
    }

    func int __inc__(fixed<uint_> me, int other) {
        copy = me
        me += 1 << uint_
        return copy
    }

    func int __dec__(fixed<uint_> me, int other) {
        copy = me
        me -= 1 << uint_
        return copy
    }

    func fixed __shr__(fixed<uint_> me, uint amount) {
        return me >> other
    }

    func fixed __shl__(fixed<uint_> me, uint amount) {
        return me << other
    }

    func fixed __mlt__(fixed<uint_> me, fixed<uint_> other) {
        return (me * other) >> uint_
    }

    func fixed __div__(fixed<uint_> me, fixed<uint_> other) {
        return (me << uint_) / other
    }

    func fixed __neg__(fixed<uint_> me) {
        return -me
    }

    func bool __gt__(fixed<uint_> me, fixed<uint_> other) {
        return me > other
    }

    func bool __gte__(fixed<uint_> me, fixed<uint_> other) {
        return me >= other
    }

    func bool __lt__(fixed<uint_> me, fixed<uint_> other) {
        return me < other
    }

    func bool __lte__(fixed<uint_> me, fixed<uint_> other) {
        return me <= other
    }

    func bool __dif__(fixed<uint_> me, fixed<uint_> other) {
        return me != other
    }

    func bool __equ__(fixed<uint_> me, fixed<uint_> other) {
        return me == other
    }
}
